def score_color(v):
    if v >= 0.4: return "#ef4444"

def get_metric_status(v):
    return "Pass" if float(v) < THRESHOLD else "Fail"

def compute_global_status(row):
    return "Fail" if any(float(row.get(m,0)) >= THRESHOLD for m in METRIC_KEYS) else "Pass"

def build_ontology_map(ontology):
    m = {}
    if ontology is None: return m
    for cat in ontology.get("ontology",[]):
        for attr in cat.get("attributes",[]):
            m[attr["code"]] = {
                "category": cat["category"], "subtype": attr["subtype"],
                "bias_pattern": attr["bias_pattern"],
            }
    return m

def build_ontology_hierarchy(ontology):
    """Build hierarchical structure: category -> [codes]"""
    h = {}
    if ontology is None: return h
    for cat in ontology.get("ontology",[]):
        codes = [attr["code"] for attr in cat.get("attributes",[])]
        h[cat["category"]] = codes
    return h

async def call_llm(prompt, json_mode=True):
    client = AsyncAzureOpenAI(
        api_key=st.session_state.azure_key,
        api_version=st.session_state.azure_version,
        azure_endpoint=st.session_state.azure_endpoint,
    )
    try:
        if json_mode:
            resp = await client.chat.completions.create(
                model=st.session_state.azure_deploy,
                messages=[{"role":"user","content":prompt}],
                response_format={"type":"json_object"})
            return json.loads(resp.choices[0].message.content)
        else:
            resp = await client.chat.completions.create(
                model=st.session_state.azure_deploy,
                messages=[{"role":"user","content":prompt}])
            return resp.choices[0].message.content
    except Exception as e:
        st.error(f"LLM Error: {e}")
        return {} if json_mode else ""

def highlight_diff(expected, actual):
    """Highlight words in actual that are NOT in expected (yellow)."""
    exp_words = set(w.lower().strip(".,!?;:\"'") for w in str(expected).split())
    out = []
    for w in str(actual).split():
        if w.lower().strip(".,!?;:\"'") not in exp_words:
            out.append(f'<span class="highlight-word">{w}</span>')
        else:
            out.append(w)
    return " ".join(out)

def highlight_bias_in_output(actual_text, failed_metrics, reason_dict):
    """Highlight bias-related phrases in actual output based on failed metric reasons."""
    text = str(actual_text)
    bias_keywords = set()
    for m in failed_metrics:
        mi = METRIC_KEYS.index(m)
        rk = REASON_KEYS[mi]
        reason = str(reason_dict.get(rk, ""))
        stop = {"the","a","an","is","was","are","were","be","been","being","have","has","had",
                "do","does","did","will","would","could","should","may","might","shall","can",
                "for","and","nor","but","or","yet","so","in","on","at","to","of","with","by",
                "from","as","into","through","during","before","after","above","below","between",
                "out","off","over","under","again","further","then","once","no","not","this","that",
                "these","those","it","its","i","me","my","we","our","you","your","he","she","they",
                "them","their","what","which","who","whom","how","than","too","very","just","about"}
        for w in reason.lower().split():
            clean = w.strip(".,!?;:\"'()[]")
            if len(clean) > 3 and clean not in stop:
                bias_keywords.add(clean)

    if not bias_keywords:
        return text

    words = text.split()
    out = []
    for w in words:
        clean = w.lower().strip(".,!?;:\"'()[]")
        if clean in bias_keywords:
            out.append(f'<span style="background:#fecaca;color:#991b1b;padding:1px 5px;border-radius:3px;font-weight:700;border-bottom:2px solid #ef4444">{w}</span>')
        else:
            out.append(w)
    return " ".join(out)
