def build_fairlearn_report(edf: pd.DataFrame) -> dict:
    """
    Builds MetricFrames using Fairlearn's own built-in metric functions.

    y_true = 1 for every row (ground truth: every test case should pass)
    y_pred = 1 if Status==Pass, 0 if Status==Fail

    Fairlearn's own metrics used directly — no custom closures:
      selection_rate         → pass rate per group
      true_positive_rate     → recall (of passing cases)
      false_positive_rate    → how often a failing case is predicted pass
      false_negative_rate    → how often a passing case is predicted fail
      true_negative_rate     → specificity
      count                  → group size

    Sensitive feature dimensions: A_Code, Category, Subtype, Bias Type
    Scalar disparity metrics: demographic_parity_difference,
                              demographic_parity_ratio,
                              equalized_odds_difference
    """
    from fairlearn.metrics import (
        MetricFrame,
        selection_rate,
        true_positive_rate,
        false_positive_rate,
        false_negative_rate,
        true_negative_rate,
        count,
        demographic_parity_difference,
        demographic_parity_ratio,
        equalized_odds_difference,
    )

    if edf is None or len(edf) == 0:
        return {}

    edf_r  = edf.reset_index(drop=True)

    # ── y_true / y_pred ───────────────────────────────────────────────────────
    # Every test case is expected to pass → y_true = 1 for all rows.
    # y_pred reflects whether the LLM output actually passed evaluation.
    y_true = np.ones(len(edf_r), dtype=int)
    y_pred = (edf_r["Status"] == "Pass").astype(int).values

    # ── Sensitive feature arrays ──────────────────────────────────────────────
    sf_code = edf_r["A_Code"].fillna("Unknown").values
    sf_cat  = edf_r["Category"].fillna("Unknown").values
    sf_sub  = edf_r["Subtype"].fillna("Unknown").values
    sf_bias = edf_r["Bias Type"].fillna("Unknown").values

    # ── Fairlearn built-in metrics dict — passed directly, no wrappers ────────
    # fail_rate is the only derived one (1 - selection_rate), kept as a lambda
    # because Fairlearn doesn't ship it by name.
    fairlearn_metrics = {
        "pass_rate"         : selection_rate,
        "fail_rate"         : lambda y_t, y_p, **kw: 1.0 - selection_rate(y_t, y_p, **kw),
        "true_positive_rate": true_positive_rate,    # TP / (TP+FN) — sensitivity
        "false_positive_rate": false_positive_rate,  # FP / (FP+TN)
        "false_negative_rate": false_negative_rate,  # FN / (FN+TP)
        "true_negative_rate": true_negative_rate,    # TN / (TN+FP) — specificity
        "count"             : count,                 # group size
    }

    # ── Build one MetricFrame per sensitive dimension ─────────────────────────
    def mf(sf):
        return MetricFrame(
            metrics           = fairlearn_metrics,
            y_true            = y_true,
            y_pred            = y_pred,
            sensitive_features= sf,
        )

    mf_code = mf(sf_code)
    mf_cat  = mf(sf_cat)
    mf_sub  = mf(sf_sub)
    mf_bias = mf(sf_bias)

    # ── Scalar disparity metrics (built-in, one call each) ────────────────────
    def scalar_disparities(sf):
        return {
            "demographic_parity_difference": demographic_parity_difference(
                y_true=y_true, y_pred=y_pred, sensitive_features=sf),
            "demographic_parity_ratio"     : demographic_parity_ratio(
                y_true=y_true, y_pred=y_pred, sensitive_features=sf),
            "equalized_odds_difference"    : equalized_odds_difference(
                y_true=y_true, y_pred=y_pred, sensitive_features=sf),
        }

    scalars_code = scalar_disparities(sf_code)
    scalars_cat  = scalar_disparities(sf_cat)
    scalars_sub  = scalar_disparities(sf_sub)
    scalars_bias = scalar_disparities(sf_bias)

    # ── Derived summary table — one row per A_Code ────────────────────────────
    bg      = mf_code.by_group.copy()
    overall = mf_code.overall.to_dict()

    summary_rows = []
    for code in bg.index:
        row = {"A_Code": code}
        for col in fairlearn_metrics:
            row[col] = round(float(bg.loc[code, col]), 4)
        ov_pr = overall.get("pass_rate", 1e-9) or 1e-9
        row["pass_rate_ratio"] = round(row["pass_rate"] / ov_pr, 3)
        row["n"] = int(bg.loc[code, "count"])
        summary_rows.append(row)
    summary_df = pd.DataFrame(summary_rows).sort_values("fail_rate", ascending=False)

    # ── Disparity table — difference between best/worst group per metric ──────
    metric_cols  = ["pass_rate","fail_rate","true_positive_rate",
                    "false_positive_rate","false_negative_rate","true_negative_rate"]
    disp_rows = []
    for m in metric_cols:
        grp   = bg[m].dropna()
        worst = grp.idxmax(); best = grp.idxmin()
        disp_rows.append({
            "Metric"      : m,
            "Worst Group" : worst,
            "Worst Value" : round(float(grp.max()), 4),
            "Best Group"  : best,
            "Best Value"  : round(float(grp.min()), 4),
            "Range"       : round(float(grp.max() - grp.min()), 4),
        })
    disparity_df = pd.DataFrame(disp_rows)

    return {
        "mf_code"       : mf_code,
        "mf_cat"        : mf_cat,
        "mf_sub"        : mf_sub,
        "mf_bias"       : mf_bias,
        "scalars_code"  : scalars_code,
        "scalars_cat"   : scalars_cat,
        "scalars_sub"   : scalars_sub,
        "scalars_bias"  : scalars_bias,
        "summary_df"    : summary_df,
        "disparity_df"  : disparity_df,
        "overall"       : overall,
        "metric_cols"   : metric_cols,
    }

def build_metrics_summary(edf: pd.DataFrame) -> list:
    summary = (edf.groupby("Category")[METRIC_COLS]
                  .mean().round(4).reset_index())
    fail_counts  = (edf[edf["Status"] == "Fail"]
                    .groupby("Category").size().reset_index(name="fail_count"))
    total_counts = edf.groupby("Category").size().reset_index(name="total_count")
    summary = summary.merge(fail_counts,  on="Category", how="left")
    summary = summary.merge(total_counts, on="Category", how="left")
    summary["fail_count"]       = summary["fail_count"].fillna(0).astype(int)
    summary["total_count"]      = summary["total_count"].fillna(0).astype(int)
    summary["failure_rate_pct"] = (
        (summary["fail_count"] / summary["total_count"] * 100).round(1).fillna(0))
    return summary.to_dict("records")


async def generate_root_causes(prompts: dict, metrics_summary: list) -> dict:
    prompt = (prompts.get("root_cause_analysis", "")
              .replace("{{metrics_summary}}", json.dumps(metrics_summary, indent=2)))
    result = await call_llm(prompt, json_mode=True)
    if not result or "root_causes" not in result:
        return {"root_causes": []}
    return result

    if not result or "recommendations" not in result:
        return {"recommendations": []}
    return result


def build_ontology_map(ontology):
    m = {}
    if ontology is None: return m
    for cat in ontology.get("ontology", []):
        for attr in cat.get("attributes", []):
            m[attr["code"]] = {"category": cat["category"], "subtype": attr["subtype"], "bias_pattern": attr["bias_pattern"]}
    return m

def build_ontology_hierarchy(ontology):
    h = {}
    if ontology is None: return h
    for cat in ontology.get("ontology", []):
        h[cat["category"]] = [attr["code"] for attr in cat.get("attributes", [])]
    return h

async def call_llm(prompt, json_mode=True):
    client = AsyncAzureOpenAI(
        api_key=st.session_state.azure_key,
        api_version=st.session_state.azure_version,
        azure_endpoint=st.session_state.azure_endpoint,
    )
    try:
        if json_mode:
            resp = await client.chat.completions.create(
                model=st.session_state.azure_deploy,
                messages=[{"role":"user","content":prompt}],
                response_format={"type":"json_object"})
            return json.loads(resp.choices[0].message.content)
        else:
            resp = await client.chat.completions.create(
                model=st.session_state.azure_deploy,
                messages=[{"role":"user","content":prompt}])
            return resp.choices[0].message.content
    except Exception as e:
        st.error(f"LLM Error: {e}")
        return {} if json_mode else ""

def highlight_diff(expected, actual):
    exp_words = set(w.lower().strip(".,!?;:\"'") for w in str(expected).split())
    return " ".join(
        f'<span class="highlight-word">{w}</span>'
        if w.lower().strip(".,!?;:\"'") not in exp_words else w
        for w in str(actual).split())

def highlight_bias_in_output(actual_text, failed_metrics, reason_dict):
    text = str(actual_text)
    bias_keywords = set()
    stop = {"the","a","an","is","was","are","were","be","been","being","have","has","had","do","does","did",
            "will","would","could","should","may","might","shall","can","for","and","nor","but","or","yet",
            "so","in","on","at","to","of","with","by","from","as","into","through","during","before","after",
            "above","below","between","out","off","over","under","again","further","then","once","no","not",
            "this","that","these","those","it","its","i","me","my","we","our","you","your","he","she","they",
            "them","their","what","which","who","whom","how","than","too","very","just","about"}
    for m in failed_metrics:
        for w in str(reason_dict.get(REASON_KEYS[METRIC_KEYS.index(m)], "")).lower().split():
            c = w.strip(".,!?;:\"'()[]")
            if len(c) > 3 and c not in stop: bias_keywords.add(c)
    if not bias_keywords: return text
    return " ".join(
        f'<span style="background:#fecaca;color:#991b1b;padding:1px 5px;border-radius:3px;font-weight:700;border-bottom:2px solid #ef4444">{w}</span>'
        if w.lower().strip(".,!?;:\"'()[]") in bias_keywords else w
        for w in text.split())

def render_pagination(current_page: int, total_pages: int, key_prefix: str):
    """Render a pagination bar and return the new page index (0-based)."""
    if total_pages <= 1:
        return current_page

    col_prev, col_pages, col_next = st.columns([1, 6, 1])

    with col_prev:
        if st.button("←", key=f"{key_prefix}_prev", disabled=(current_page == 0), use_container_width=True):
            return max(0, current_page - 1)

    with col_pages:
        # Show up to 7 page buttons with ellipsis
        def page_buttons(n_pages, cur):
            pages = []
            if n_pages <= 7:
                pages = list(range(n_pages))
            else:
                if cur <= 3:
                    pages = list(range(5)) + ["...", n_pages - 1]
                elif cur >= n_pages - 4:
                    pages = [0, "..."] + list(range(n_pages - 5, n_pages))
                else:
                    pages = [0, "...", cur - 1, cur, cur + 1, "...", n_pages - 1]
            return pages

        btns = page_buttons(total_pages, current_page)
        cols = st.columns(len(btns))
        for i, (col, p) in enumerate(zip(cols, btns)):
            with col:
                if p == "...":
                    st.markdown('<div style="text-align:center;padding:6px 0;color:#94a3b8;font-size:13px">…</div>', unsafe_allow_html=True)
                else:
                    is_active = (p == current_page)
                    btn_style = (
                        "background:#2563eb;color:white;border:none;" if is_active
                        else "background:white;color:#475569;border:1px solid #e2e8f0;"
                    )
                    if st.button(
                        str(p + 1),
                        key=f"{key_prefix}_pg_{p}",
                        use_container_width=True,
                        disabled=is_active,
                    ):
                        return p

    with col_next:
        if st.button("→", key=f"{key_prefix}_next", disabled=(current_page == total_pages - 1), use_container_width=True):
            return min(total_pages - 1, current_page + 1)

    return current_page
