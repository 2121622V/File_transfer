def score_color(v):
    if v >= 0.4: return "#ef4444"
    if v >= 0.2: return "#f59e0b"
    return "#10b981"

def pct(v): return f"{v*100:.1f}%"

def build_ontology_map(ontology):
    m = {}
    for cat in ontology.get("ontology", []):
        for attr in cat.get("attributes", []):
            m[attr["code"]] = {
                "category":    cat["category"],
                "subtype":     attr["subtype"],
                "bias_pattern":attr["bias_pattern"],
            }
    return m

async def call_llm(prompt, json_mode=True):
    from openai import AsyncAzureOpenAI
    client = AsyncAzureOpenAI(
        api_key      = st.session_state.azure_key,
        api_version  = st.session_state.azure_version,
        azure_endpoint=st.session_state.azure_endpoint,
    )
    try:
        if json_mode:
            resp = await client.chat.completions.create(
                model=st.session_state.azure_deploy,
                messages=[{"role":"user","content":prompt}],
                response_format={"type":"json_object"}
            )
            return json.loads(resp.choices[0].message.content)
        else:
            resp = await client.chat.completions.create(
                model=st.session_state.azure_deploy,
                messages=[{"role":"user","content":prompt}]
            )
            return resp.choices[0].message.content
    except Exception as e:
        st.error(f"LLM Error: {e}")
        return {} if json_mode else ""

def get_metric_status(v):
    return "Pass" if float(v) < THRESHOLD else "Fail"

def compute_global_status(row):
    vals = [row.get(m, 0.0) for m in METRIC_KEYS]
    return "Fail" if any(float(v) >= THRESHOLD for v in vals) else "Pass"

        if st.button("▶ Run Evaluation", type="primary"):
            eval_rows = []
            progress  = st.progress(0, text="Evaluating…")

            async def eval_one(row):
                eval_prompt = (prompts.get("evaluation","")
                               .replace("{{baseline}}", str(row["expected_output"]))
                               .replace("{{persona}}",  str(row["Actual Output"])))
                return await call_llm(eval_prompt)

            async def run_eval():
                tasks = [eval_one(row) for _, row in gen_df.iterrows()]
                return await asyncio.gather(*tasks, return_exceptions=True)

            results = asyncio.run(run_eval())
            progress.empty()

            eval_df = gen_df.copy()
            for col in METRIC_KEYS + REASON_KEYS + STATUS_KEYS:
                eval_df[col] = None

            key_map = {
                "Decision Flip":        ("decision_flip_score",        "decision_flip_reason"),
                "Fairness Degradation": ("fairness_degradation_score", "fairness_degradation_reason"),
                "Tone Inequality":      ("tone_inequality_score",       "tone_inequality_reason"),
                "Safety_Drift":         ("safety_drift_score",          "safety_drift_reason"),
                "Stereotype Injection": ("stereotype_injection_score",  "stereotype_injection_reason"),
            }

            for i, (res, (_, row)) in enumerate(zip(results, gen_df.iterrows())):
                progress.progress((i+1)/len(gen_df), text=f"Processing {i+1}/{len(gen_df)}…")
                if isinstance(res, Exception) or not res:
                    for m in METRIC_KEYS:
                        eval_df.at[row.name, m] = 0.0
                    continue
                for m, (score_k, reason_k) in key_map.items():
                    eval_df.at[row.name, m] = float(res.get(score_k, 0.0))
                eval_df.at[row.name, "Reason_Flip"]     = res.get("decision_flip_reason",        "No issue detected.")
                eval_df.at[row.name, "Reason_Fairness"] = res.get("fairness_degradation_reason", "No issue detected.")
                eval_df.at[row.name, "Reason_Tone"]     = res.get("tone_inequality_reason",       "No issue detected.")
                eval_df.at[row.name, "Reason_Safety"]   = res.get("safety_drift_reason",          "No issue detected.")
                eval_df.at[row.name, "Reason_Stereo"]   = res.get("stereotype_injection_reason",  "No issue detected.")

            progress.empty()

            for m in METRIC_KEYS:
                eval_df[m] = pd.to_numeric(eval_df[m], errors="coerce").fillna(0.0)

            for mk, sk in zip(METRIC_KEYS, STATUS_KEYS):
                eval_df[sk] = eval_df[mk].apply(get_metric_status)
            eval_df["Status"] = eval_df.apply(compute_global_status, axis=1)
            eval_df.insert(0, "id", range(1, len(eval_df)+1))

            eval_df.to_excel(OUTPUT_DIR / "bias_results.xlsx", index=False)

            st.success(f"✅ Evaluation complete — saved to Updated Output/bias_results.xlsx")
