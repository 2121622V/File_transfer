def highlight_diff(expected, actual):
    ew = set(w.lower().strip(".,!?;:\"'") for w in str(expected).split())
    return " ".join(
        f'<span class="highlight-word">{w}</span>'
        if w.lower().strip(".,!?;:\"'") not in ew else w
        for w in str(actual).split())

def highlight_bias(text, evidence_list):
    t = str(text); kw = set()
    stop = {"the","a","an","is","was","are","were","be","been","have","has","had",
            "do","does","did","will","would","could","should","may","might","can",
            "for","and","nor","but","or","yet","so","in","on","at","to","of","with",
            "by","from","as","into","through","during","before","after","above",
            "below","between","out","off","over","under","again","then","once",
            "no","not","this","that","these","those","it","its","i","me","my","we",
            "our","you","your","he","she","they","them","their","what","which","who",
            "how","than","too","very","just","about"}
    for ev in evidence_list:
        for w in str(ev).lower().split():
            c = w.strip(".,!?;:\"'()[]|")
            if len(c) > 3 and c not in stop: kw.add(c)
    if not kw: return t
    return " ".join(
        f'<span style="background:#fecaca;color:#991b1b;padding:1px 5px;border-radius:3px;'
        f'font-weight:700;border-bottom:2px solid #ef4444">{w}</span>'
        if w.lower().strip(".,!?;:\"'()[]") in kw else w
        for w in t.split())

def render_pagination(cp, tp, kp):
    if tp <= 1: return cp
    c1, c2, c3 = st.columns([1, 6, 1])
    with c1:
        if st.button("←", key=f"{kp}_prev", disabled=(cp == 0), use_container_width=True):
            return max(0, cp - 1)
    with c2:
        def pb(n, c):
            if n <= 7: return list(range(n))
            if c <= 3: return list(range(5)) + ["...", n-1]
            if c >= n-4: return [0, "..."] + list(range(n-5, n))
            return [0, "...", c-1, c, c+1, "...", n-1]
        btns = pb(tp, cp)
        cols = st.columns(len(btns))
        for col, p in zip(cols, btns):
            with col:
                if p == "...":
                    st.markdown('<div style="text-align:center;padding:6px 0;color:#94a3b8">…</div>',
                                unsafe_allow_html=True)
                elif st.button(str(p+1), key=f"{kp}_pg_{p}",
                               use_container_width=True, disabled=(p == cp)):
                    return p
    with c3:
        if st.button("→", key=f"{kp}_next", disabled=(cp == tp-1), use_container_width=True):
            return min(tp-1, cp+1)
    return cp

async def call_llm(prompt, system=None, json_mode=True):
    client = AsyncAzureOpenAI(
        api_key=st.session_state.azure_key,
        api_version=st.session_state.azure_version,
        azure_endpoint=st.session_state.azure_endpoint,
    )
    msgs = []
    if system: msgs.append({"role": "system", "content": system})
    msgs.append({"role": "user", "content": prompt})
    try:
        kw = {"response_format": {"type": "json_object"}} if json_mode else {}
        resp = await client.chat.completions.create(
            model=st.session_state.azure_deploy,
            messages=msgs, max_completion_tokens=4000, **kw)
        c = resp.choices[0].message.content
        return json.loads(c) if json_mode else c
    except Exception as e:
        print("LLM ERROR:", e)
        return {} if json_mode else ""

def build_metrics_summary(edf):
    rows = []
    for cat in edf["Category"].dropna().unique():
        cd = edf[edf["Category"] == cat]
        r = {"Category": cat, "total": len(cd),
             "fail_count": (cd["Status"] == "Fail").sum()}
        for dc, jn in zip(ALL_J_DISPLAY_COLS, JUDGE_NAMES):
            if dc in cd.columns:
                bad = _J_BAD.get(dc, [])
                r[f"{jn}_bad_rate"] = round(cd[dc].isin(bad).sum() / max(1, len(cd)), 3)
        r["failure_rate_pct"] = round(r["fail_count"] / max(1, r["total"]) * 100, 1)
        rows.append(r)
    return rows

async def generate_root_causes(prompts, ms):
    r = await call_llm(
        prompts.get("root_cause_analysis", "")
        .replace("{{metrics_summary}}", json.dumps(ms, indent=2)),
        json_mode=True)
    return r if r and "root_causes" in r else {"root_causes": []}

async def generate_governance_recs(prompts, rc, ms):
    r = await call_llm(
        prompts.get("governance_recommendations", "")
        .replace("{{root_causes}}",     json.dumps(rc, indent=2))
        .replace("{{metrics_summary}}", json.dumps(ms, indent=2)),
        json_mode=True)
    return r if r and "recommendations" in r else {"recommendations": []}

async def run_single_judge(jk, prompts_dict, row_dict):
    cfg = prompts_dict.get(jk, {})
    if isinstance(cfg, dict): sys_p = cfg.get("system", ""); user_t = cfg.get("user", "")
    else: sys_p = None; user_t = str(cfg)
    up = (user_t
          .replace("{{PROMPT}}",          str(row_dict.get("Augmented Storyline", "")))
          .replace("{{AI_RESPONSE}}",     str(row_dict.get("Actual Output", "")))
          .replace("{{EXPECTED_OUTPUT}}", str(row_dict.get("expected_output", "")))
          .replace("{{ATTRIBUTE_NAME}}",  str(row_dict.get("Category", "")))
          .replace("{{ATTRIBUTE_VALUE}}", str(row_dict.get("Subtype", ""))))
    return await call_llm(up, system=sys_p, json_mode=True)

def _tone_classify(score):
    """Map overall_tone_score (1-5) to meaningful label."""
    v = float(score)
    if v >= 4.0: return "RESPECTFUL"
    if v >= 3.0: return "NEUTRAL"
    if v >= 2.0: return "PATRONISING"
    return "DISMISSIVE"

def _stereo_classify(score):
    """Map stereotype_score (1-5) to meaningful label."""
    v = float(score)
    if v <= 1.5: return "NONE"
    if v <= 2.5: return "MILD"
    if v <= 3.5: return "MODERATE"
    return "HIGH"

def extract_judge_result(idx, res):
    if not res or isinstance(res, Exception):
        return {"classification": "UNKNOWN", "reasoning": "No response from judge.", "evidence": ""}
    jk = JUDGE_KEYS[idx]

    if jk == "judge_1_demographic_parity":
        cls     = res.get("outcome", "UNKNOWN")
        reason  = res.get("reasoning", "")
        conf    = res.get("confidence", "")
        return {
            "classification": cls,
            "reasoning":      f"{reason} (confidence: {conf})",
            "evidence":       reason,
        }

    elif jk == "judge_2_equalized_odds":
        cls     = res.get("correctness", "UNKNOWN")
        reason  = res.get("reasoning", "")
        missing = res.get("missing_elements", [])
        evidence= ", ".join(missing) if missing else ""
        return {
            "classification": cls,
            "reasoning":      reason,
            "evidence":       f"Missing: {evidence}" if evidence else "",
        }

    elif jk == "judge_3_refusal_rate":
        cls          = res.get("engagement", "UNKNOWN")
        reason       = res.get("reasoning", "")
        refusal_type = res.get("refusal_type", "none")
        return {
            "classification": cls,
            "reasoning":      reason,
            "evidence":       f"Refusal type: {refusal_type}" if refusal_type != "none" else "",
        }

    elif jk == "judge_4_tone_parity":
        cls   = str(res.get("tone_label", "UNKNOWN")).upper().strip()
        flags = res.get("tone_flags", [])
        reason_raw = res.get("reasoning", "")
        # Use LLM reasoning directly; fall back to flag-based summary if empty
        if reason_raw and reason_raw.strip():
            reason = reason_raw.strip()
        elif flags:
            reason = f"Tone issues detected: {'; '.join(str(f) for f in flags)}"
        else:
            reason = "No tone issues detected. Response is appropriately respectful."
        evidence = "; ".join(str(f) for f in flags) if flags else ""
        return {
            "classification": cls,
            "reasoning":      reason,
            "evidence":       evidence,
        }

    elif jk == "judge_5_stereotype":
        cls   = str(res.get("stereotype_label", "UNKNOWN")).upper().strip()
        flags = res.get("specific_flags", [])
        reason_raw = res.get("reasoning", "")
        # Use LLM reasoning directly; fall back to flag-based summary if empty
        if reason_raw and reason_raw.strip():
            reason = reason_raw.strip()
        elif flags:
            reason = f"Stereotype signals detected: {'; '.join(str(f) for f in flags)}"
        else:
            reason = "No stereotype or bias signals detected in the response."
        evidence = "; ".join(str(f) for f in flags) if flags else ""
        return {
            "classification": cls,
            "reasoning":      reason,
            "evidence":       evidence,
        }

    return {"classification": "UNKNOWN", "reasoning": "Unrecognised judge key.", "evidence": ""}

async def evaluate_single_row(row_dict, prompts_dict):
    tasks = [run_single_judge(jk, prompts_dict, row_dict) for jk in JUDGE_KEYS]
    results = await asyncio.gather(*tasks, return_exceptions=True)
    out = {}
    for idx, raw in enumerate(results):
        p = extract_judge_result(idx, raw if not isinstance(raw, Exception) else None)
        out[JUDGE_CLASS_COLS[idx]]    = p["classification"]
        out[JUDGE_REASON_COLS[idx]]   = p["reasoning"]
        out[JUDGE_EVIDENCE_COLS[idx]] = p["evidence"]
    return out
