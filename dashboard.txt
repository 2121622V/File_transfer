def build_ontology_map(ontology):

    for cat in ontology.get("ontology", []):

    return m

def highlight_diff(expected, actual):
    ew = set(w.lower().strip(".,!?;:\"'") for w in str(expected).split())
    return " ".join(
        f'<span class="highlight-word">{w}</span>'
        if w.lower().strip(".,!?;:\"'") not in ew else w
        for w in str(actual).split())

def highlight_bias(text, evidence_list):
    t = str(text); kw = set()
    stop = {"the","a","an","is","was","are","were","be","been","have","has","had","do","does","did","will","would","could","should","may","might","can","for","and","nor","but","or","yet","so","in","on","at","to","of","with","by","from","as","into","through","during","before","after","above","below","between","out","off","over","under","again","then","once","no","not","this","that","these","those","it","its","i","me","my","we","our","you","your","he","she","they","them","their","what","which","who","how","than","too","very","just","about"}
    for ev in evidence_list:
        for w in str(ev).lower().split():
            c = w.strip(".,!?;:\"'()[]|")
            if len(c) > 3 and c not in stop: kw.add(c)
    if not kw: return t
    return " ".join(
        f'<span style="background:#fecaca;color:#991b1b;padding:1px 5px;border-radius:3px;font-weight:700;border-bottom:2px solid #ef4444">{w}</span>'
        if w.lower().strip(".,!?;:\"'()[]") in kw else w
        for w in t.split())

def render_pagination(cp, tp, kp):
    if tp <= 1: return cp
    c1, c2, c3 = st.columns([1, 6, 1])
    with c1:
        if st.button("←", key=f"{kp}_prev", disabled=(cp == 0), use_container_width=True): return max(0, cp - 1)
    with c2:
        def pb(n, c):
            if n <= 7: return list(range(n))
            if c <= 3: return list(range(5)) + ["...", n-1]
            if c >= n-4: return [0, "..."] + list(range(n-5, n))
            return [0, "...", c-1, c, c+1, "...", n-1]
        btns = pb(tp, cp)
        cols = st.columns(len(btns))
        for col, p in zip(cols, btns):
            with col:
                if p == "...":
                    st.markdown('<div style="text-align:center;padding:6px 0;color:#94a3b8">…</div>', unsafe_allow_html=True)
                elif st.button(str(p+1), key=f"{kp}_pg_{p}", use_container_width=True, disabled=(p == cp)):
                    return p
    with c3:
        if st.button("→", key=f"{kp}_next", disabled=(cp == tp-1), use_container_width=True): return min(tp-1, cp+1)
    return cp

async def call_llm(prompt, system=None, json_mode=True):
    client = AsyncAzureOpenAI(
        api_key        = st.session_state.azure_key,
        api_version    = st.session_state.azure_version,
        azure_endpoint = st.session_state.azure_endpoint,
    )
    msgs = []
    if system:
        msgs.append({"role": "system", "content": system})
    msgs.append({"role": "user", "content": prompt})

    try:
        kwargs = {}
        if json_mode:
            kwargs["response_format"] = {"type": "json_object"}

        resp = await client.chat.completions.create(
            model                  = st.session_state.azure_deploy,
            messages               = msgs,
            max_completion_tokens  = 4000,
            **kwargs
        )
        raw = resp.choices[0].message.content or ""

        if not json_mode:
            return raw

        cleaned = raw.strip()
        if cleaned.startswith("```"):
            lines   = cleaned.split("\n")
            cleaned = "\n".join(lines[1:] if len(lines) > 1 else lines)
        if cleaned.endswith("```"):
            lines   = cleaned.split("\n")
            cleaned = "\n".join(lines[:-1] if len(lines) > 1 else lines)
        cleaned = cleaned.strip()

        if not cleaned:
            st.session_state["_llm_last_error"] = "LLM returned empty response body."
            return {}

        try:
            return json.loads(cleaned)
        except json.JSONDecodeError as je:
            err = f"JSON parse error: {je} | Raw (first 400 chars): {cleaned[:400]}"
            st.session_state["_llm_last_error"] = err
            return {}

    except Exception as e:
        err = f"{type(e).__name__}: {e}"
        st.session_state["_llm_last_error"] = err
        return {} if json_mode else ""

async def run_single_judge(jk, prompts_dict, row_dict):
    cfg = prompts_dict.get(jk, {})
    if isinstance(cfg, dict): sys_p = cfg.get("system", ""); user_t = cfg.get("user", "")
    else: sys_p = None; user_t = str(cfg)
    up = (user_t
          .replace("{{PROMPT}}",          str(row_dict.get("Augmented Storyline", "")))
          .replace("{{AI_RESPONSE}}",     str(row_dict.get("Actual Output", "")))
          .replace("{{EXPECTED_OUTPUT}}", str(row_dict.get("expected_output", "")))
          .replace("{{ATTRIBUTE_NAME}}",  str(row_dict.get("Category", "")))
          .replace("{{ATTRIBUTE_VALUE}}", str(row_dict.get("Subtype", ""))))
    return await call_llm(up, system=sys_p, json_mode=True)

def extract_judge_result(idx, res):
    if not res or isinstance(res, Exception):
        return {"classification": "UNKNOWN", "reasoning": "Judge returned no response.", "evidence": ""}
    jk = JUDGE_KEYS[idx]

    if jk == "judge_1_demographic_parity":
        cls     = str(res.get("outcome", "UNKNOWN")).upper().strip()
        reason  = res.get("reasoning", "")
        conf    = res.get("confidence", "")
        phrases = res.get("evidence_phrases", [])
        evidence = "; ".join(str(p) for p in phrases) if phrases else ""
        return {"classification": cls, "reasoning": f"{reason} (confidence: {conf})" if conf else reason, "evidence": evidence}

    elif jk == "judge_2_equalized_odds":
        cls     = str(res.get("correctness", "UNKNOWN")).upper().strip()
        reason  = res.get("reasoning", "")
        missing = res.get("missing_elements", [])
        evidence = "Missing: " + "; ".join(str(m) for m in missing) if missing else ""
        return {"classification": cls, "reasoning": reason, "evidence": evidence}

    elif jk == "judge_3_refusal_rate":
        cls          = str(res.get("engagement", "UNKNOWN")).upper().strip()
        reason       = res.get("reasoning", "")
        refusal_type = res.get("refusal_type", "none")
        phrases      = res.get("evidence_phrases", [])
        evidence     = "; ".join(str(p) for p in phrases) if phrases else (
                       f"Refusal type: {refusal_type}" if refusal_type and refusal_type != "none" else "")
        return {"classification": cls, "reasoning": reason, "evidence": evidence}

    elif jk == "judge_4_tone_parity":
        cls        = str(res.get("tone_label", "UNKNOWN")).upper().strip()
        flags      = res.get("tone_flags", [])
        reason_raw = res.get("reasoning", "")
        if reason_raw and reason_raw.strip():
            reason = reason_raw.strip()
        elif flags:
            reason = f"Tone issues detected: {'; '.join(str(f) for f in flags)}"
        else:
            reason = "No tone issues detected. Response is appropriately respectful."
        evidence = "; ".join(str(f) for f in flags) if flags else ""
        return {"classification": cls, "reasoning": reason, "evidence": evidence}

    elif jk == "judge_5_stereotype":
        cls        = str(res.get("stereotype_label", "UNKNOWN")).upper().strip()
        flags      = res.get("specific_flags", [])
        reason_raw = res.get("reasoning", "")
        if reason_raw and reason_raw.strip():
            reason = reason_raw.strip()
        elif flags:
            reason = f"Stereotype signals detected: {'; '.join(str(f) for f in flags)}"
        else:
            reason = "No stereotype or bias signals detected in the response."
        evidence = "; ".join(str(f) for f in flags) if flags else ""
        return {"classification": cls, "reasoning": reason, "evidence": evidence}

    return {"classification": "UNKNOWN", "reasoning": "Unrecognised judge.", "evidence": ""}

async def evaluate_single_row(row_dict, prompts_dict):
    tasks   = [run_single_judge(jk, prompts_dict, row_dict) for jk in JUDGE_KEYS]
    results = await asyncio.gather(*tasks, return_exceptions=True)
    out = {}
    for idx, raw in enumerate(results):
        p = extract_judge_result(idx, raw if not isinstance(raw, Exception) else None)
        out[JUDGE_CLASS_COLS[idx]]    = p["classification"]
        out[JUDGE_REASON_COLS[idx]]   = p["reasoning"]
        out[JUDGE_EVIDENCE_COLS[idx]] = p["evidence"]
    return out

                    "Judge":    jname,
                    "Category": cat,
                    "Subtype":  sub,
                    "Total":    n_sub,
